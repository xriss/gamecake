// Callback function
static cpBool PlaySoundOnImpact(cpArbiter *arb, cpSpace *space, void *data){
    PlayCrashSound();
    return cpTrue;
}

{
    ...

    handler->postSolveFunc = PlaySoundOnImpact;

}
